    # debounce

    实现防抖

    <template>
  <!-- 定义count累加按钮 -->
  <div>
    <button @click="addCount">{{count}}</button>
  </div>
</template>

<script lang='ts' setup>
// 引入lodash
import lodash from 'lodash'
// 引入ref
import { ref } from 'vue'
// 定义count数据
const count = ref(0)
// 定义addCount方法，并进行防抖处理
const addCount = lodash.debounce(() => {
  count.value++
}, 1000)
</script>
